## Problem 1
What will the following code print?

```js
const hour = 10;

function amOrPm() {
    if (hour >= 0 && hour < 12) {
        return "am";
    } else if (hour >= 12 && hour < 24)  {
        return "pm";
    } else {
        return "invalid time"
    }
}

console.log(amOrPm(10));
console.log(amOrPm(14));
console.log(amOrPm(20));
console.log(amOrPm(25));
```
